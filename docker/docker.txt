1.版本
docker version
2.查询信息
docker info
3.运行
docker run hello-world
4.镜像列表
docker images
docker images -a //显示全部,和不加参数基本一致
docker images -q //只显示id
docker images --digests//显示hash值
docker images --no-trunc//完整信息
docker search tomcat //查询tomcat镜像信息
docker search -s 30 tomcat
docker pull tomcat //下载tomcat镜像
docker rmi tomcat //删除镜像
docker rmi `docker images -q` //删除所有镜像
docker build -f hellodockerfile -t hello .
5.容器列表
docker ps
docker ps -l
docker start id//启动容器
docker stop id//停止容器
docker restart id//重启容器
docker run -d id //后台运行
docker exec -t 容器id
docker run -d -p8888:8080 镜像id //后台端口映射
docker commit 容器id 
docker cp 容器id:容器内文件 路径//复制文件到操作系统
docker rm 容器id //删除容器
docker rm `docker ps -a -q` //删除所有容器
